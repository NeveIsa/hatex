\begin{homeworkSection}{1a}
\problemAnswer{
	$f(x)=atan(x)$ on interval $[a,b] = [-4.9,5.1]$
	
	$f(a)=atan(-4.9) = 0.7854$ and $f(b) = atan(5.1)=-1.3695$
	
	Since $atan(x)\ \epsilon\ C[-4.9, 5.1]$ and $f(-4.9)f(5.1)<0$ the conditions required for bisection method
	to converge are satisfied.
	
	
	The Number of iterations is given by $M = \ceil*{log_2(\frac{b-a}{2\delta})}$ where $\delta$ = Absolute error = $10^-2$
	
	Hence $M=\ceil*{log_2(\frac{10}{2*10^-2})} = 9$
	
	
	}	
\end{homeworkSection}

\begin{homeworkSection}{4}

\problemAnswer{
	$g(x)=x+cf(x)$\\
	$g'(x)=1+cf'(x)$\\
	$g''(x)=cf''(x)$\\
	
	Sufficent condition for convergence of $x_{n+1}=g(x_n)$ : $|g'(x)|<1$ for x
	
	so $|1+cf'(x)| < 1$ 
	
	When $1+cf'(x)>0$,  $c<\frac{1}{f'(x)}$ or $c<min(\frac{1}{f'(x)})$ for x in  domain(f)
	
	
	
	}

\end{homeworkSection}
\begin{homeworkSection}{5}
	\problemAnswer{
		$g(x)=\frac{1}{3}(\frac{x^3}{3}-x^2-\frac{5}{4}x+4)$\\
		$g'(x)=\frac{1}{3}(x^2-2x+1-\frac{3^2}{2^2})$\\
		$g'(x)=\frac{1}{3}(x-1+1.5)(x-1-1.5)$\\
		$g'(x)=\frac{1}{3}(x-0.5)(x-2.5)$
		$g''(x) =\frac{1}{3}(2x-1)$
		$g''(x)<0$ for $x \in [0,0.5]$
		and $g''(x)>0$ for $x \in [0.5,2]$
		
		Thus,$g'(x)$ is bounded between $max=g'(0)$ and $min=g'(2)$
		
		$max=g'(0)=1.25/3$ and $min=-0.5$ and hence $|g'(x)|<1$ for $x \in [0,2]$
		
		Thus, by contraction mapping theorem $g(x)$ is convergent(it maps to itself and is continuous)
		
	}
\end{homeworkSection}
